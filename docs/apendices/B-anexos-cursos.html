tabla_final <- (cc_facso %>% filter(ano != "2026")) %>%
  group_by(ano, carrera) %>%
  summarise(total_cursos = n_distinct(codigo, seccion), .groups = "drop") %>%  # <-- aquí
  pivot_wider(
    names_from  = carrera,
    values_from = total_cursos,
    values_fill = 0
  ) %>%
  mutate(ano = as.character(ano)) %>%
  bind_rows(
    summarise(., across(-ano, sum)) %>% mutate(ano = "Total")
  ) %>%
  mutate(Total = rowSums(across(-ano)))

# Nombres de columnas: "Año" + nombres reales de carreras + "Total"
nombres_columnas <- c("Año", names(tabla_final)[-1])

tabla_final %>%
  kable(col.names = nombres_columnas) %>%
  kable_styling(
    bootstrap_options = c("striped", "hover", "condensed", "responsive"),
    full_width = FALSE,
    position = "center"
  ) %>%
  row_spec(
    0,
    bold = TRUE,
    color = "white",
    background = "#952626"
  ) %>%
  row_spec(
    nrow(tabla_final),
    bold = TRUE,

  ) %>%

#| label: tbl-n-notas

#| tbl-cap-location: top

tabla_notas_n <- cc_facso %>%
  mutate(
    cat_estudiantes = case_when(
      n_estudiantes == 1                          ~ "1 estudiante",
      n_estudiantes > 1  & n_estudiantes <= 5      ~ "> 1 y <= 5",
      n_estudiantes > 5  & n_estudiantes <= 10    ~ "> 5 y <= 10",
      n_estudiantes > 10 & n_estudiantes <= 20   ~ "> 10 y <= 20",
      n_estudiantes > 20  & n_estudiantes <= 40   ~ "> 20 y <= 40",
      n_estudiantes > 40                          ~ "> 40"
    ),
    cat_estudiantes = factor(cat_estudiantes,
  levels = c("1 estudiante", "> 1 y <= 5", "> 5 y <= 10",
             "> 10 y <= 20", "> 20 y <= 40", "> 40"))
  ) %>%
  group_by(cat_estudiantes) %>%
  summarise(
    frec_absoluta = n(),
    nota_promedio = mean(nota_promedio, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  mutate(
    frec_relativa  = frec_absoluta / sum(frec_absoluta),
    nota_prom_acum = cumsum(nota_promedio * frec_absoluta) / cumsum(frec_absoluta)
  ) %>%
  select(cat_estudiantes, frec_absoluta, frec_relativa, nota_promedio, nota_prom_acum)

tabla_notas_n %>%
  bind_rows(
    summarise(.,
      cat_estudiantes = factor("Total"),
      frec_absoluta   = sum(frec_absoluta),
      frec_relativa   = sum(frec_relativa),
      nota_promedio   = sum(nota_promedio * frec_absoluta) / sum(frec_absoluta),
      nota_prom_acum  = max(nota_prom_acum)
    )
  ) %>%
  kable(
    digits = 2,
    col.names = c(
      "Tamaño del curso",
      "Frec. absoluta",
      "Frec. relativa",
      "Nota promedio",
      "Nota prom. acumulada"
    )
  ) %>%
  kable_styling(
    bootstrap_options = c("striped", "hover", "condensed", "responsive"),
    full_width = FALSE,
    position = "center"
  ) %>%
  row_spec(
    0,
    bold = TRUE,
    color = "white",
    background = "#952626"
  ) %>%
  row_spec(
    nrow(tabla_notas_n) + 1,
    bold = TRUE,
    background = "#f2f2f2"
  )

tabla_notas_ano <- cc_facso %>%
  mutate(
    cat_nota = case_when(
      nota_promedio < 4.0                        ~ "< 4,0",
      nota_promedio >= 4.0 & nota_promedio < 5.0 ~ ">= 4,0 y < 5,0",
      nota_promedio >= 5.0 & nota_promedio < 6.0 ~ ">= 5,0 y < 6,0",
      nota_promedio >= 6.0 & nota_promedio < 6.5 ~ ">= 6,0 y < 6,5",
      nota_promedio >= 6.5                       ~ ">= 6,5"
    ),
    cat_nota = factor(cat_nota,
      levels = c("< 4,0", ">= 4,0 y < 5,0", ">= 5,0 y < 6,0", ">= 6,0 y < 6,5", ">= 6,5"))
  ) %>%
  group_by(cat_nota) %>%
  summarise(
    frec_absoluta = n(),
    .groups = "drop"
  ) %>%
  mutate(
    frec_relativa  = frec_absoluta / sum(frec_absoluta),
    frec_acumulada = cumsum(frec_relativa)
  )

tabla_notas_ano %>%

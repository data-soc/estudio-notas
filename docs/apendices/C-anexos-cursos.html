tabla_final <- (cc_facso %>% filter(ano != "2026")) %>%
  group_by(ano, carrera) %>%
  summarise(total_cursos = n_distinct(codigo, seccion), .groups = "drop") %>%  # <-- aquí
  pivot_wider(
    names_from  = carrera,
    values_from = total_cursos,
    values_fill = 0
  ) %>%
  mutate(ano = as.character(ano)) %>%
  bind_rows(
    summarise(., across(-ano, sum)) %>% mutate(ano = "Total")
  ) %>%
  mutate(Total = rowSums(across(-ano)))

tabla_notas_n <- cc_facso %>%
  mutate(
    cat_estudiantes = case_when(
      n_estudiantes == 1             ~ "1 estudiante",
      n_estudiantes > 1 & n_estudiantes < 5  ~ "> 1 y < 5",
      n_estudiantes >= 5 & n_estudiantes < 10 ~ ">= 5 y < 10",
      n_estudiantes >= 10            ~ ">= 10"
    ),
    cat_estudiantes = factor(cat_estudiantes,
      levels = c("1 estudiante", "> 1 y < 5", ">= 5 y < 10", ">= 10"))
  ) %>%
  group_by(cat_estudiantes) %>%
  summarise(
    frec_absoluta   = n(),
    nota_promedio   = mean(nota_promedio, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  mutate(
    frec_relativa   = frec_absoluta / sum(frec_absoluta),
    nota_prom_acum  = cumsum(nota_promedio * frec_absoluta) / cumsum(frec_absoluta)
  ) %>%
  select(
    cat_estudiantes,
    frec_absoluta,
    frec_relativa,
    nota_promedio,
    nota_prom_acum
  )

tabla_notas_n %>%
  kable(
    digits = 2,
    col.names = c(
      "Tamaño del curso",
      "Frec. absoluta",
      "Frec. relativa",
      "Nota promedio",
      "Nota prom. acumulada"
    )
  ) %>%
  kable_styling(
    bootstrap_options = c("striped", "hover", "condensed", "responsive"),
    full_width = FALSE,
    position = "center"
  ) %>%
  row_spec(
    0,
    bold = TRUE,
    color = "white",
    background = "#952626"
  ) %>%
  row_spec(
    nrow(tabla_notas_n),
    bold = TRUE,
    background = "#f2f2f2"
  )
